00032     //  Only a single thread can read from the pipe at any specific moment.
00033     //  Only a single thread can write to the pipe at any specific moment.

00099         //  Flush all the completed items into the pipe. Returns false if
00100         //  the reader thread is sleeping. In that case, caller is obliged to
00101         //  wake the reader up before using the pipe again.
00102         inline bool flush ()
00103         {
00104             //  If there are no un-flushed items, do nothing.
00105             if (w == f)
00106                 return true;
00107 
00108             //  Try to set 'c' to 'f'.
00109             if (c.cas (w, f) != w) {

00111                 //  Compare-and-swap was unseccessful because 'c' is NULL.

00113                 //  care about thread-safeness and update c in non-atomic
00114                 //  manner. We'll return false to let the caller know
00115                 //  that reader is sleeping.

00122             //  the 'first un-flushed item' pointer to 'f'.

00129         {
