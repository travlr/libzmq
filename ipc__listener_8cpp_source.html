00071 {
00072     fd_t fd = accept ();

00074     //  If connection was reset by the peer in the meantime, just ignore it.
00075     //  TODO: Handle specific errors like ENFILE/EMFILE etc.
00076     if (fd == retired_fd)

00080     stream_engine_t *engine = new (std::nothrow) stream_engine_t (fd, options);

00083     //  Choose I/O thread to run connecter in. Given that we are already

00088     //  Create and launch a session object.

00098 {
00099     //  Get rid of the file associated with the UNIX domain socket that
00100     //  may have been left behind by the previous run of the application.
00101     ::unlink (addr_);
00102     filename.clear ();
00103 
00104     //  Initialise the address structure.
00105     ipc_address_t address;
00106     int rc = address.resolve (addr_);

00108         return -1;
00109 
00110     //  Create a listening socket.

00149 zmq::fd_t zmq::ipc_listener_t::accept ()
00150 {
00151     //  Accept one connection and deal with different failure modes.
00152     zmq_assert (s != retired_fd);
00153     fd_t sock = ::accept (s, NULL, NULL);
00154     if (sock == -1) {
