00181 int zmq::vtcp_connecter_t::get_new_reconnect_ivl ()
00182 {
00183     //  The new interval is the current interval + random value.
00184     int this_interval = current_reconnect_ivl +
00185         (generate_random () % options.reconnect_ivl);
00186 
00187     //  Only change the current reconnect interval  if the maximum reconnect
00188     //  interval was set and if it's larger than the reconnect interval.
00189     if (options.reconnect_ivl_max > 0 && 
00190         options.reconnect_ivl_max > options.reconnect_ivl) {
00191 
00192         //  Calculate the next interval
00193         current_reconnect_ivl = current_reconnect_ivl * 2;
00194         if(current_reconnect_ivl >= options.reconnect_ivl_max) {
00195             current_reconnect_ivl = options.reconnect_ivl_max;
00196         }   
00197     }
00198     return this_interval;
