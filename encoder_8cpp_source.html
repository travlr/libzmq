00055 bool zmq::encoder_t::message_ready ()
00056 {
00057     //  Destroy content of the old message.
00058     int rc = in_progress.close ();
00059     errno_assert (rc == 0);
00060 
00061     //  Read new message. If there is none, return false.
00062     //  Note that new state is set only if write is successful. That way
00063     //  unsuccessful write will cause retry on the next state machine
00064     //  invocation.
00065     if (!source || !source->read (&in_progress)) {
00066         rc = in_progress.init ();
00067         errno_assert (rc == 0);
00068         return false;
00069     }
00070 
00071     //  Get the message size.
00072     size_t size = in_progress.size ();
00073 
00074     //  Account for the 'flags' byte.
00075     size++;
00076 
00077     //  For messages less than 255 bytes long, write one byte of message size.
00078     //  For longer messages write 0xff escape character followed by 8-byte
00079     //  message size. In both cases 'flags' field follows.
00080     if (size < 255) {
00081         tmpbuf [0] = (unsigned char) size;
00082         tmpbuf [1] = (in_progress.flags () & ~msg_t::shared);
00083         next_step (tmpbuf, 2, &encoder_t::size_ready,
00084             !(in_progress.flags () & (msg_t::more | msg_t::label)));
00085     }
00086     else {
00087         tmpbuf [0] = 0xff;
00088         put_uint64 (tmpbuf + 1, size);
00089         tmpbuf [9] = (in_progress.flags () & ~msg_t::shared);
00090         next_step (tmpbuf, 10, &encoder_t::size_ready,
00091             !(in_progress.flags () & (msg_t::more | msg_t::label)));
00092     }
00093     return true;
00094 }
