00124     int rc = address.resolve (addr_, true, options.ipv4only ? true : false);
00125     if (rc != 0)

00136     if (address.family () == AF_INET6 && errno == EAFNOSUPPORT &&
00137           !options.ipv4only) {

00139         if (rc != 0)
00140             return rc;

00142     }

00144 #ifdef ZMQ_HAVE_WINDOWS
00145     if (s == INVALID_SOCKET) {
00146         wsa_error_to_errno ();

00148     }
00149 #else

00151         return -1;

00153 
00154     //  On some systems, IPv4 mapping in IPv6 sockets is disabled by default.
00155     //  Switch it on in such cases.
00156     if (address.family () == AF_INET6)
00157         enable_ipv4_mapping (s);
00158 
00159     //  Allow reusing of the address.
00160     int flag = 1;
00161 #ifdef ZMQ_HAVE_WINDOWS
00162     rc = setsockopt (s, SOL_SOCKET, SO_EXCLUSIVEADDRUSE,
00163         (const char*) &flag, sizeof (int));
00164     wsa_assert (rc != SOCKET_ERROR);
00165 #else
00166     rc = setsockopt (s, SOL_SOCKET, SO_REUSEADDR, &flag, sizeof (int));
00167     errno_assert (rc == 0);
00168 #endif
00169 
00170     //  Bind the socket to the network interface and port.
00171     rc = bind (s, address.addr (), address.addrlen ());
00172 #ifdef ZMQ_HAVE_WINDOWS
00173     if (rc == SOCKET_ERROR) {
00174         wsa_error_to_errno ();

00184 #ifdef ZMQ_HAVE_WINDOWS
