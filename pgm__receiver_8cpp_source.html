00111 void zmq::pgm_receiver_t::activate_in ()
00112 {
00113     //  It is possible that the most recently used decoder
00114     //  processed the whole buffer but failed to write
00115     //  the last message into the pipe.
00116     if (pending_bytes == 0) {
00117         if (mru_decoder != NULL)
00118             mru_decoder->process_buffer (NULL, 0);
00119         return;
00120     }
00121 
00122     zmq_assert (mru_decoder != NULL);
00123     zmq_assert (pending_ptr != NULL);
00124 
00125     //  Ask the decoder to process remaining data.
00126     size_t n = mru_decoder->process_buffer (pending_ptr, pending_bytes);
00127     pending_bytes -= n;
00128 
00129     if (pending_bytes > 0)
00130         return;
00131 
00132     //  Resume polling.
00133     set_pollin (pipe_handle);
00134     set_pollin (socket_handle);
00135 
00136     in_event ();
00137 }

00139 void zmq::pgm_receiver_t::in_event ()
00140 {
00141     // Read data from the underlying pgm_socket.
00142     unsigned char *data = NULL;
00143     const pgm_tsi_t *tsi = NULL;
00144 
00145     zmq_assert (pending_bytes == 0);
00146 
00147     if (has_rx_timer) {
00148         cancel_timer (rx_timer_id);
00149         has_rx_timer = false;
00150     }
00151 
00152     //  TODO: This loop can effectively block other engines in the same I/O
00153     //  thread in the case of high load.
00154     while (true) {
00155 
00156         //  Get new batch of data.
00157         //  Note the workaround made not to break strict-aliasing rules.
00158         void *tmp = NULL;
00159         ssize_t received = pgm_socket.receive (&tmp, &tsi);
00160         data = (unsigned char*) tmp;
00161 
00162         //  No data to process. This may happen if the packet received is
00163         //  neither ODATA nor ODATA.
00164         if (received == 0) {
00165             if (errno == ENOMEM || errno == EBUSY) {
00166                 const long timeout = pgm_socket.get_rx_timeout ();
00167                 add_timer (timeout, rx_timer_id);
00168                 has_rx_timer = true;
00169             }
00170             break;
00171         }
00172 
00173         //  Find the peer based on its TSI.
00174         peers_t::iterator it = peers.find (*tsi);
00175 
00176         //  Data loss. Delete decoder and mark the peer as disjoint.
00177         if (received == -1) {
00178             if (it != peers.end ()) {
00179                 it->second.joined = false;
00180                 if (it->second.decoder == mru_decoder)
00181                     mru_decoder = NULL;
00182                 if (it->second.decoder != NULL) {
00183                     delete it->second.decoder;
00184                     it->second.decoder = NULL;
00185                 }
00186             }
00187             break;
00188         }
00189 
00190         //  New peer. Add it to the list of know but unjoint peers.
00191         if (it == peers.end ()) {
00192             peer_info_t peer_info = {false, NULL};
00193             it = peers.insert (peers_t::value_type (*tsi, peer_info)).first;
00194         }
00195 
00196         //  Read the offset of the fist message in the current packet.
00197         zmq_assert ((size_t) received >= sizeof (uint16_t));
00198         uint16_t offset = get_uint16 (data);
00199         data += sizeof (uint16_t);
00200         received -= sizeof (uint16_t);
00201 
00202         //  Join the stream if needed.
00203         if (!it->second.joined) {
00204 
00205             //  There is no beginning of the message in current packet.
00206             //  Ignore the data.
00207             if (offset == 0xffff)
00208                 continue;
00209 
00210             zmq_assert (offset <= received);
00211             zmq_assert (it->second.decoder == NULL);
00212 
00213             //  We have to move data to the begining of the first message.
00214             data += offset;
00215             received -= offset;
00216 
00217             //  Mark the stream as joined.
00218             it->second.joined = true;
00219 
00220             //  Create and connect decoder for the peer.
00221             it->second.decoder = new (std::nothrow) decoder_t (0,
00222                 options.maxmsgsize);
00223             alloc_assert (it->second.decoder);
00224             it->second.decoder->set_session (session);
00225         }
00226 
00227         mru_decoder = it->second.decoder;
00228 
00229         //  Push all the data to the decoder.
00230         ssize_t processed = it->second.decoder->process_buffer (data, received);
00231         if (processed < received) {
00232             //  Save some state so we can resume the decoding process later.
00233             pending_bytes = received - processed;
00234             pending_ptr = data + processed;
00235             //  Stop polling.
00236             reset_pollin (pipe_handle);
00237             reset_pollin (socket_handle);
00238 
00239             //  Reset outstanding timer.
00240             if (has_rx_timer) {
00241                 cancel_timer (rx_timer_id);
00242                 has_rx_timer = false;
00243             }
00244 
00245             break;
00246         }
00247     }
00248 
00249     //  Flush any messages decoder may have produced.
00250     session->flush ();
00251 }
