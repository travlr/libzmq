00036     //  Helper base class for decoders that know the amount of data to read
00037     //  in advance at any moment. Knowing the amount in advance is a property
00038     //  of the protocol used. 0MQ framing protocol is based size-prefixed
00039     //  paradigm, whixh qualifies it to be parsed by this class.
00040     //  On the other hand, XML-based transports (like XMPP or SOAP) don't allow
00041     //  for knowing the size of data to read in advance and should use different
00042     //  decoding algorithms.
00043     //
00044     //  This class implements the state machine that parses the incoming buffer.
00045     //  Derived class should implement individual state machine actions.
00046

00048     {
00049     public:

00068         //  Returns a buffer to be filled with binary data.
00069         inline void get_buffer (unsigned char **data_, size_t *size_)
00070         {
00071             //  If we are expected to read large message, we'll opt for zero-
00072             //  copy, i.e. we'll ask caller to fill the data directly to the
00073             //  message. Note that subsequent read(s) are non-blocking, thus
00074             //  each single read reads at most SO_RCVBUF bytes at once not
00075             //  depending on how large is the chunk returned from here.
00076             //  As a consequence, large messages being received won't block
00077             //  other engines running in the same I/O thread for excessive
00078             //  amounts of time.
00079             if (to_read >= bufsize) {
00080                 *data_ = read_pos;
00081                 *size_ = to_read;
00082                 return;
00083             }
00084 
00085             *data_ = buf;
00086             *size_ = bufsize;
00087         }
00088 
00089         //  Processes the data in the buffer previously allocated using
00090         //  get_buffer function. size_ argument specifies nemuber of bytes
00091         //  actually filled into the buffer. Function returns number of
00092         //  bytes actually processed.
00093         inline size_t process_buffer (unsigned char *data_, size_t size_)
00094         {
00095             //  Check if we had an error in previous attempt.
00096             if (unlikely (!(static_cast <T*> (this)->next)))
00097                 return (size_t) -1;
00098 
00099             //  In case of zero-copy simply adjust the pointers, no copying
00100             //  is required. Also, run the state machine in case all the data
00101             //  were processed.
00102             if (data_ == read_pos) {
00103                 read_pos += size_;
00104                 to_read -= size_;
00105 
00106                 while (!to_read) {
00107                     if (!(static_cast <T*> (this)->*next) ()) {
00108                         if (unlikely (!(static_cast <T*> (this)->next)))
00109                             return (size_t) -1;
00110                         return size_;
00111                     }
00112                 }
00113                 return size_;
00114             }
00115 
00116             size_t pos = 0;
00117             while (true) {
00118 
00119                 //  Try to get more space in the message to fill in.
00120                 //  If none is available, return.
00121                 while (!to_read) {
00122                     if (!(static_cast <T*> (this)->*next) ()) {
00123                         if (unlikely (!(static_cast <T*> (this)->next)))
00124                             return (size_t) -1;
00125                         return pos;
00126                     }
00127                 }
00128 
00129                 //  If there are no more data in the buffer, return.
00130                 if (pos == size_)
00131                     return pos;
00132 
00133                 //  Copy the data from buffer to the message.
00134                 size_t to_copy = std::min (to_read, size_ - pos);
00135                 memcpy (read_pos, data_ + pos, to_copy);
00136                 read_pos += to_copy;
00137                 pos += to_copy;
00138                 to_read -= to_copy;
00139             }
00140         }
