00173 void zmq::mailbox_t::send (const command_t &cmd_)
00174 {
00175     //  Attempt to write an entire command without blocking.
00176     ssize_t nbytes;
00177     do {
00178         nbytes = ::send (w, &cmd_, sizeof (command_t), 0);
00179     } while (nbytes == -1 && errno == EINTR);
00180 
00181     //  Attempt to increase mailbox SNDBUF if the send failed.
00182     if (nbytes == -1 && errno == EAGAIN) {
00183         int old_sndbuf, new_sndbuf;
00184         socklen_t sndbuf_size = sizeof old_sndbuf;
00185 
00186         //  Retrieve current send buffer size.
00187         int rc = getsockopt (w, SOL_SOCKET, SO_SNDBUF, &old_sndbuf,
00188             &sndbuf_size);
00189         errno_assert (rc == 0);
00190         new_sndbuf = old_sndbuf * 2;
00191 
00192         //  Double the new send buffer size.
00193         rc = setsockopt (w, SOL_SOCKET, SO_SNDBUF, &new_sndbuf, sndbuf_size);
00194         errno_assert (rc == 0);
00195 
00196         //  Verify that the OS actually honored the request.
00197         rc = getsockopt (w, SOL_SOCKET, SO_SNDBUF, &new_sndbuf, &sndbuf_size);
00198         errno_assert (rc == 0);
00199         zmq_assert (new_sndbuf > old_sndbuf);
00200 
00201         //  Retry the sending operation; at this point it must succeed.
00202         do {
00203             nbytes = ::send (w, &cmd_, sizeof (command_t), 0);
00204         } while (nbytes == -1 && errno == EINTR);
00205     }
00206     errno_assert (nbytes != -1);
00207 
00208     //  This should never happen as we've already checked that command size is
00209     //  less than PIPE_BUF.
00210     zmq_assert (nbytes == sizeof (command_t));
00211 }
