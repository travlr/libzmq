00054 int zmq::mailbox_t::recv (command_t *cmd_, int timeout_)
00055 {
00056     //  Try to get the command straight away.
00057     if (active) {
00058         bool ok = cpipe.read (cmd_);
00059         if (ok)
00060             return 0;
00061 
00062         //  If there are no more commands available, switch into passive state.
00063         active = false;
00064         signaler.recv ();
00065     }
00066 
00067     //  Wait for signal from the command sender.
00068     int rc = signaler.wait (timeout_);
00069     if (rc != 0 && (errno == EAGAIN || errno == EINTR))
00070         return -1;

00072     //  We've got the signal. Now we can switch into active state.
00073     active = true;
