00157 void zmq::mailbox_t::send (const command_t &cmd_)
00158 {
00159     //  Attempt to write an entire command without blocking.
00160     ssize_t nbytes;
00161     do {
00162         nbytes = ::send (w, &cmd_, sizeof (command_t), 0);
00163     } while (nbytes == -1 && errno == EINTR);
00164 
00165     //  Attempt to increase mailbox SNDBUF if the send failed.
00166     if (nbytes == -1 && errno == EAGAIN) {
00167         int old_sndbuf, new_sndbuf;
00168         socklen_t sndbuf_size = sizeof old_sndbuf;
00169 
00170         //  Retrieve current send buffer size.
00171         int rc = getsockopt (w, SOL_SOCKET, SO_SNDBUF, &old_sndbuf,
00172             &sndbuf_size);
00173         errno_assert (rc == 0);
00174         new_sndbuf = old_sndbuf * 2;
00175 
00176         //  Double the new send buffer size.
00177         rc = setsockopt (w, SOL_SOCKET, SO_SNDBUF, &new_sndbuf, sndbuf_size);
00178         errno_assert (rc == 0);
00179 
00180         //  Verify that the OS actually honored the request.
00181         rc = getsockopt (w, SOL_SOCKET, SO_SNDBUF, &new_sndbuf, &sndbuf_size);
00182         errno_assert (rc == 0);
00183         zmq_assert (new_sndbuf > old_sndbuf);
00184 
00185         //  Retry the sending operation; at this point it must succeed.
00186         do {
00187             nbytes = ::send (w, &cmd_, sizeof (command_t), 0);
00188         } while (nbytes == -1 && errno == EINTR);
00189     }
00190     errno_assert (nbytes != -1);
00191 
00192     //  This should never happen as we've already checked that command size is
00193     //  less than PIPE_BUF.
00194     zmq_assert (nbytes == sizeof (command_t));
