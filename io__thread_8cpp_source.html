00064 void zmq::io_thread_t::in_event ()
00065 {
00066     //  TODO: Do we want to limit number of commands I/O thread can
00067     //  process in a single go?
00068 
00069     while (true) {
00070 
00071         //  Get the next command. If there is none, exit.
00072         command_t cmd;
00073         int rc = mailbox.recv (&cmd, false);
00074         if (rc != 0 && errno == EINTR)
00075             continue;
00076         if (rc != 0 && errno == EAGAIN)
00077             break;
00078         errno_assert (rc == 0);
00079 
00080         //  Process the command.
00081         cmd.destination->process_command (cmd);
00082     }
00083 }
