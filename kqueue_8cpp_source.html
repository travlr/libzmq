00146 void zmq::kqueue_t::loop ()
00147 {
00148     while (!stopping) {
00149 
00150         //  Execute any due timers.
00151         int timeout = (int) execute_timers ();
00152 
00153         //  Wait for events.
00154         struct kevent ev_buf [max_io_events];
00155         timespec ts = {timeout / 1000, (timeout % 1000) * 1000000};
00156         int n = kevent (kqueue_fd, NULL, 0, &ev_buf [0], max_io_events,
00157             timeout ? &ts: NULL);
00158         if (n == -1 && errno == EINTR)
00159             continue;
00160         errno_assert (n != -1);
00161 
00162         for (int i = 0; i < n; i ++) {
00163             poll_entry_t *pe = (poll_entry_t*) ev_buf [i].udata;
00164 
00165             if (pe->fd == retired_fd)
00166                 continue;
00167             if (ev_buf [i].flags & EV_EOF)
00168                 pe->reactor->in_event ();
00169             if (pe->fd == retired_fd)
00170                 continue;
00171             if (ev_buf [i].filter == EVFILT_WRITE)
00172                 pe->reactor->out_event ();
00173             if (pe->fd == retired_fd)
00174                 continue;
00175             if (ev_buf [i].filter == EVFILT_READ)
00176                 pe->reactor->in_event ();
00177         }
00178 
00179         //  Destroy retired event sources.
00180         for (retired_t::iterator it = retired.begin (); it != retired.end ();
00181               ++it)
00182             delete *it;
00183         retired.clear ();
00184     }
00185 }
