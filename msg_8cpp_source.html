00104 int zmq::msg_t::close ()
00105 {
00106     //  Check the validity of the message.
00107     if (unlikely (!check ())) {
00108         errno = EFAULT;
00109         return -1;
00110     }
00111 
00112     if (u.base.type == type_lmsg) {
00113 
00114         //  If the content is not shared, or if it is shared and the reference
00115         //  count has dropped to zero, deallocate it.
00116         if (!(u.lmsg.flags & msg_t::shared) ||
00117               !u.lmsg.content->refcnt.sub (1)) {
00118 
00119             //  We used "placement new" operator to initialize the reference
00120             //  counter so we call the destructor explicitly now.
00121             u.lmsg.content->refcnt.~atomic_counter_t ();
00122 
00123             if (u.lmsg.content->ffn)
00124                 u.lmsg.content->ffn (u.lmsg.content->data,
00125                     u.lmsg.content->hint);
00126             free (u.lmsg.content);
00127         }
00128     }
00129 
00130     //  Make the message invalid.
00131     u.base.type = 0;
