00036     //  fills the outgoing buffer. Derived classes should implement individual
00037     //  state machine actions.

00041     public:

00057         //  The function returns a batch of binary data. The data
00058         //  are filled to a supplied buffer. If no buffer is supplied (data_
00059         //  points to NULL) decoder object will provide buffer of its own.
00060         //  If offset is not NULL, it is filled by offset of the first message
00061         //  in the batch.If there's no beginning of a message in the batch,
00062         //  offset is set to -1.
00063         inline void get_data (unsigned char **data_, size_t *size_,
00064             int *offset_ = NULL)
00065         {
00066             unsigned char *buffer = !*data_ ? buf : *data_;
00067             size_t buffersize = !*data_ ? bufsize : *size_;
00068 
00069             size_t pos = 0;
00070             if (offset_)
00071                 *offset_ = -1;
00072 
00073             while (true) {
00074 
00075                 //  If there are no more data to return, run the state machine.
00076                 //  If there are still no data, return what we already have
00077                 //  in the buffer.
00078                 if (!to_write) {
00079                     if (!(static_cast <T*> (this)->*next) ()) {
00080                         *data_ = buffer;
00081                         *size_ = pos;
00082                         return;
00083                     }
00084 
00085                     //  If beginning of the message was processed, adjust the
00086                     //  first-message-offset.
00087                     if (beginning) { 
00088                         if (offset_ && *offset_ == -1)
00089                             *offset_ = (int) pos;
00090                         beginning = false;
00091                     }
00092                 }
00093 
00094                 //  If there are no data in the buffer yet and we are able to
00095                 //  fill whole buffer in a single go, let's use zero-copy.
00096                 //  There's no disadvantage to it as we cannot stuck multiple
00097                 //  messages into the buffer anyway. Note that subsequent
00098                 //  write(s) are non-blocking, thus each single write writes
00099                 //  at most SO_SNDBUF bytes at once not depending on how large
00100                 //  is the chunk returned from here.
00101                 //  As a consequence, large messages being sent won't block
00102                 //  other engines running in the same I/O thread for excessive
00103                 //  amounts of time.
00104                 if (!pos && !*data_ && to_write >= buffersize) {
00105                     *data_ = write_pos;
00106                     *size_ = to_write;
00107                     write_pos = NULL;
00108                     to_write = 0;
00109                     return;
00110                 }
00111 
00112                 //  Copy data to the buffer. If the buffer is full, return.
00113                 size_t to_copy = std::min (to_write, buffersize - pos);
00114                 memcpy (buffer + pos, write_pos, to_copy);
00115                 pos += to_copy;
00116                 write_pos += to_copy;
00117                 to_write -= to_copy;
00118                 if (pos == buffersize) {
00119                     *data_ = buffer;
00120                     *size_ = pos;
00121                     return;
00122                 }
00123             }
00124         }
