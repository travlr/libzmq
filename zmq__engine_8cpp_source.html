00104 void zmq::zmq_engine_t::in_event ()
00105 {
00106     bool disconnection = false;
00107 
00108     //  If there's no data to process in the buffer...
00109     if (!insize) {
00110 
00111         //  Retrieve the buffer and read as much data as possible.
00112         //  Note that buffer can be arbitrarily large. However, we assume
00113         //  the underlying TCP layer has fixed buffer size and thus the
00114         //  number of bytes read will be always limited.
00115         decoder.get_buffer (&inpos, &insize);
00116         insize = tcp_socket.read (inpos, insize);
00117 
00118         //  Check whether the peer has closed the connection.
00119         if (insize == (size_t) -1) {
00120             insize = 0;
00121             disconnection = true;
00122         }
00123     }
00124 
00125     //  Push the data to the decoder.
00126     size_t processed = decoder.process_buffer (inpos, insize);
00127 
00128     if (unlikely (processed == (size_t) -1)) {
00129         disconnection = true;
00130     }
00131     else {
00132 
00133         //  Stop polling for input if we got stuck.
00134         if (processed < insize) {
00135 
00136             //  This may happen if queue limits are in effect or when
00137             //  init object reads all required information from the socket
00138             //  and rejects to read more data.
00139             if (plugged)
00140                 reset_pollin (handle);
00141         }
00142 
00143         //  Adjust the buffer.
00144         inpos += processed;
00145         insize -= processed;
00146     }
00147 
00148     //  Flush all messages the decoder may have produced.
00149     //  If IO handler has unplugged engine, flush transient IO handler.
00150     if (unlikely (!plugged)) {
00151         zmq_assert (ephemeral_sink);
00152         ephemeral_sink->flush ();
00153     } else {
00154         sink->flush ();
00155     }
00156 
00157     if (sink && disconnection)
00158         error ();
00159 }

00161 void zmq::zmq_engine_t::out_event ()
00162 {
00163     //  If write buffer is empty, try to read new data from the encoder.
00164     if (!outsize) {
00165 
00166         outpos = NULL;
00167         encoder.get_data (&outpos, &outsize);
00168 
00169         //  If IO handler has unplugged engine, flush transient IO handler.
00170         if (unlikely (!plugged)) {
00171             zmq_assert (ephemeral_sink);
00172             ephemeral_sink->flush ();
00173             return;
00174         }
00175 
00176         //  If there is no data to send, stop polling for output.
00177         if (outsize == 0) {
00178             reset_pollout (handle);
00179             return;
00180         }
00181     }
00182 
00183     //  If there are any data to write in write buffer, write as much as
00184     //  possible to the socket. Note that amount of data to write can be
00185     //  arbitratily large. However, we assume that underlying TCP layer has
00186     //  limited transmission buffer and thus the actual number of bytes
00187     //  written should be reasonably modest.
00188     int nbytes = tcp_socket.write (outpos, outsize);
00189 
00190     //  Handle problems with the connection.
00191     if (nbytes == -1) {
00192         error ();
00193         return;
00194     }
00195 
00196     outpos += nbytes;
00197     outsize -= nbytes;
00198 }
