00084     //  Set POLLIN. We wont never want to stop polling for uplink = we never

00086     set_pollin (uplink_handle);
00087     set_pollin (rdata_notify_handle);
00088     set_pollin (pending_notify_handle);
00089 
00090     //  Set POLLOUT for downlink_socket_handle.
00091     set_pollout (handle);
00092 
00093     //  PGM is not able to pass subscriptions upstream, thus we have no idea
00094     //  what messages are peers interested in. Because of that we have to
00095     //  subscribe for all the messages.
00096     msg_t msg;
00097     msg.init_size (1);

00100     zmq_assert (ok);

00149 {
00150     if (has_rx_timer) {

00164 void zmq::pgm_sender_t::out_event ()
00165 {
00166     //  POLLOUT event from send socket. If write buffer is empty, 
00167     //  try to read new data from the encoder.
00168     if (write_size == 0) {
00169 
00170         //  First two bytes (sizeof uint16_t) are used to store message 
00171         //  offset in following steps. Note that by passing our buffer to
00172         //  the get data function we prevent it from returning its own buffer.
00173         unsigned char *bf = out_buffer + sizeof (uint16_t);
00174         size_t bfsz = out_buffer_size - sizeof (uint16_t);
00175         int offset = -1;
00176         encoder.get_data (&bf, &bfsz, &offset);
00177 
00178         //  If there are no data to write stop polling for output.
00179         if (!bfsz) {
00180             reset_pollout (handle);
00181             return;
00182         }
00183 
00184         //  Put offset information in the buffer.
00185         write_size = bfsz + sizeof (uint16_t);
00186         put_uint16 (out_buffer, offset == -1 ? 0xffff : (uint16_t) offset);
00187     }
00188 
00189     if (has_tx_timer) {
00190         cancel_timer (tx_timer_id);
00191         has_tx_timer = false;
00192     }
00193 
00194     //  Send the data.
00195     size_t nbytes = pgm_socket.send (out_buffer, write_size);
00196 
00197     //  We can write either all data or 0 which means rate limit reached.
00198     if (nbytes == write_size) {
00199         write_size = 0;
00200     } else {
00201         zmq_assert (nbytes == 0);
00202 
00203         if (errno == ENOMEM) {
00204             const long timeout = pgm_socket.get_tx_timeout ();
00205             add_timer (timeout, tx_timer_id);
00206             has_tx_timer = true;
00207         } else
00208             zmq_assert (errno == EBUSY);
00209     }
00210 }
