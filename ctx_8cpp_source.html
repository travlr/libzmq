00116 {

00118     //  restarted.

00121     slot_sync.unlock ();

00123     //  First attempt to terminate the context.
00124     if (!restarted) {
00125 
00126         //  Close the logging infrastructure.

00134         //  interrupted. If there are no sockets we can ask reaper thread to stop.
00135         slot_sync.lock ();
