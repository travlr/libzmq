00075 void zmq::zmq_connecter_t::out_event ()
00076 {
00077     fd_t fd = tcp_connecter.connect ();
00078     rm_fd (handle);
00079     handle_valid = false;
00080 
00081     //  Handle the error condition by attempt to reconnect.
00082     if (fd == retired_fd) {
00083         tcp_connecter.close ();
00084         wait = true;
00085         add_reconnect_timer();
00086         return;
00087     }
00088 
00089     //  Choose I/O thread to run connecter in. Given that we are already
00090     //  running in an I/O thread, there must be at least one available.
00091     io_thread_t *io_thread = choose_io_thread (options.affinity);
00092     zmq_assert (io_thread);
00093 
00094     //  Create an init object. 
00095     zmq_init_t *init = new (std::nothrow) zmq_init_t (io_thread, NULL,

00098     launch_sibling (init);
00099 
00100     //  Shut the connecter down.

00111 void zmq::zmq_connecter_t::start_connecting ()
00112 {
00113     //  Open the connecting socket.
00114     int rc = tcp_connecter.open ();
00115 
00116     //  Connect may succeed in synchronous manner.
00117     if (rc == 0) {
00118         handle = add_fd (tcp_connecter.get_fd ());
00119         handle_valid = true;
00120         out_event ();
00121         return;
00122     }
00123 
00124     //  Connection establishment may be dealyed. Poll for its completion.
00125     else if (rc == -1 && errno == EAGAIN) {
00126         handle = add_fd (tcp_connecter.get_fd ());
00127         handle_valid = true;
00128         set_pollout (handle);
00129         return;
00130     }
00131 
00132     //  Handle any other error condition by eventual reconnect.
00133     wait = true;
00134     add_reconnect_timer();
00135 }
