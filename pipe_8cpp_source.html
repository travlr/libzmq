00288 void zmq::create_pipe (object_t *reader_parent_, object_t *writer_parent_,
00289     int hwm_, reader_t **reader_, writer_t **writer_)
00290 {
00291     //  First compute the low water mark. Following point should be taken
00292     //  into consideration:
00293     //
00294     //  1. LWM has to be less than HWM.
00295     //  2. LWM cannot be set to very low value (such as zero) as after filling
00296     //     the queue it would start to refill only after all the messages are
00297     //     read from it and thus unnecessarily hold the progress back.
00298     //  3. LWM cannot be set to very high value (such as HWM-1) as it would
00299     //     result in lock-step filling of the queue - if a single message is
00300     //     read from a full queue, writer thread is resumed to write exactly one
00301     //     message to the queue and go back to sleep immediately. This would
00302     //     result in low performance.
00303     //
00304     //  Given the 3. it would be good to keep HWM and LWM as far apart as
00305     //  possible to reduce the thread switching overhead to almost zero,
00306     //  say HWM-LWM should be max_wm_delta.
00307     //
00308     //  That done, we still we have to account for the cases where
00309     //  HWM < max_wm_delta thus driving LWM to negative numbers.
00310     //  Let's make LWM 1/2 of HWM in such cases.
00311     int lwm = (hwm_ > max_wm_delta * 2) ?
00312         hwm_ - max_wm_delta : (hwm_ + 1) / 2;
00313 
00314     //  Create all three objects pipe consists of: the pipe per se, reader and
00315     //  writer. The pipe will be handled by reader and writer, its never passed
00316     //  to the user. Reader and writer are returned to the user.
00317     pipe_t *pipe = new (std::nothrow) pipe_t ();
00318     alloc_assert (pipe);
00319     *reader_ = new (std::nothrow) reader_t (reader_parent_, pipe, lwm);
00320     alloc_assert (*reader_);
00321     *writer_ = new (std::nothrow) writer_t (writer_parent_, pipe, *reader_,
00322         hwm_);
00323     alloc_assert (*writer_);
00324 }
