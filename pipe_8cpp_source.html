00233 void zmq::pipe_t::process_pipe_term ()
00234 {
00235     //  This is the simple case of peer-induced termination. If there are no
00236     //  more pending messages to read, or if the pipe was configured to drop
00237     //  pending messages, we can move directly to the terminating state.
00238     //  Otherwise we'll hang up in pending state till all the pending messages
00239     //  are sent.
00240     if (state == active) {
00241         if (!delay) {
00242             state = terminating;
00243             outpipe = NULL;
00244             send_pipe_term_ack (peer);
00245             return;
00246         }
00247         else {
00248             state = pending;
00249             return;
00250         }
00251     }
00252 
00253     //  Delimiter happened to arrive before the term command. Now we have the
00254     //  term command as well, so we can move straight to terminating state.
00255     if (state == delimited) {
00256         state = terminating;
00257         outpipe = NULL;
00258         send_pipe_term_ack (peer);
00259         return;
00260     }
00261 
00262     //  This is the case where both ends of the pipe are closed in parallel.
00263     //  We simply reply to the request by ack and continue waiting for our
00264     //  own ack.
00265     if (state == terminated) {
00266         state = double_terminated;
00267         outpipe = NULL;
00268         send_pipe_term_ack (peer);
00269         return;
00270     }
00271 
00272     //  pipe_term is invalid in other states.
00273     zmq_assert (false);

00276 void zmq::pipe_t::process_pipe_term_ack ()
00277 {
00278     //  Notify the user that all the references to the pipe should be dropped.
00279     zmq_assert (sink);
00280     sink->terminated (this);
00281 
00282     //  In terminating and double_terminated states there's nothing to do.
00283     //  Simply deallocate the pipe. In terminated state we have to ack the
00284     //  peer before deallocating this side of the pipe. All the other states
00285     //  are invalid.
00286     if (state == terminating) ;
00287     else if (state == double_terminated);
00288     else if (state == terminated) {
00289         outpipe = NULL;
00290         send_pipe_term_ack (peer);
00291     }
00292     else
00293         zmq_assert (false);
00294 
00295     //  We'll deallocate the inbound pipe, the peer will deallocate the outbound
00296     //  pipe (which is an inbound pipe from its point of view).
00297     //  First, delete all the unread messages in the pipe. We have to do it by
00298     //  hand because msg_t doesn't have automatic destructor. Then deallocate
00299     //  the ypipe itself.
00300     msg_t msg;
00301     while (inpipe->read (&msg)) {
00302        int rc = msg.close ();
00303        errno_assert (rc == 0);

00305     delete inpipe;
00306

00311 void zmq::pipe_t::terminate (bool delay_)
00312 {
00313     //  Overload the value specified at pipe creation.
00314     delay = delay_;
00315 
00316     //  If terminate was already called, we can ignore the duplicit invocation.
00317     if (state == terminated || state == double_terminated)
00318         return;
00319 
00320     //  If the pipe is in the final phase of async termination, it's going to
00321     //  closed anyway. No need to do anything special here.
00322     else if (state == terminating)
00323         return;
00324 
00325     //  The simple sync termination case. Ask the peer to terminate and wait
00326     //  for the ack.
00327     else if (state == active) {
00328         send_pipe_term (peer);
00329         state = terminated;
00330     }
00331 
00332     //  There are still pending messages available, but the user calls
00333     //  'terminate'. We can act as if all the pending messages were read.
00334     else if (state == pending && !delay) {
00335             outpipe = NULL;
00336             send_pipe_term_ack (peer);
00337             state = terminating;
00338     }
00339 
00340     //  If there are pending messages still availabe, do nothing.
00341     else if (state == pending && delay) {
00342     }
00343 
00344     //  We've already got delimiter, but not term command yet. We can ignore
00345     //  the delimiter and ack synchronously terminate as if we were in
00346     //  active state.    
00347     else if (state == delimited) {
00348         send_pipe_term (peer);
00349         state = terminated;    
00350     }
00351 
00352     //  There are no other states.
00353     else
00354         zmq_assert (false);
00355 
00356     //  Stop outbound flow of messages.
00357     out_active = false;
00358 
00359     if (outpipe) {
00360 
00361         //  Rollback any unfinished outbound messages.
00362         rollback ();
00363 
00364         //  Push delimiter into the outbound pipe. Note that watermarks are not
00365         //  checked thus the delimiter can be written even though the pipe is full.
00366         msg_t msg;
00367         msg.init_delimiter ();
00368         outpipe->write (msg, false);

00370     }

00378 int zmq::pipe_t::compute_lwm (int hwm_)
00379 {
00380     //  Compute the low water mark. Following point should be taken
00381     //  into consideration:
00382     //
00383     //  1. LWM has to be less than HWM.
00384     //  2. LWM cannot be set to very low value (such as zero) as after filling
00385     //     the queue it would start to refill only after all the messages are
00386     //     read from it and thus unnecessarily hold the progress back.
00387     //  3. LWM cannot be set to very high value (such as HWM-1) as it would
00388     //     result in lock-step filling of the queue - if a single message is
00389     //     read from a full queue, writer thread is resumed to write exactly one
00390     //     message to the queue and go back to sleep immediately. This would
00391     //     result in low performance.
00392     //
00393     //  Given the 3. it would be good to keep HWM and LWM as far apart as
00394     //  possible to reduce the thread switching overhead to almost zero,
00395     //  say HWM-LWM should be max_wm_delta.
00396     //
00397     //  That done, we still we have to account for the cases where
00398     //  HWM < max_wm_delta thus driving LWM to negative numbers.
00399     //  Let's make LWM 1/2 of HWM in such cases.
00400     int result = (hwm_ > max_wm_delta * 2) ?
00401         hwm_ - max_wm_delta : (hwm_ + 1) / 2;
00402 
00403     return result;
00404 }
