00116 int zmq::xrep_t::xsend (msg_t *msg_, int flags_)
00117 {
00118     //  If this is the first part of the message it's the identity of the
00119     //  peer to send the message to.
00120     if (!more_out) {
00121         zmq_assert (!current_out);
00122 
00123         //  If we have malformed message (prefix with no subsequent message)
00124         //  then just silently ignore it.
00125         if (msg_->flags () & msg_t::more) {
00126 
00127             more_out = true;
00128 
00129             //  Find the pipe associated with the identity stored in the prefix.
00130             //  If there's no such pipe just silently ignore the message.
00131             blob_t identity ((unsigned char*) msg_->data (), msg_->size ());
00132             outpipes_t::iterator it = outpipes.find (identity);
00133 
00134             if (it != outpipes.end ()) {
00135                 current_out = it->second.pipe;
00136                 msg_t empty;
00137                 int rc = empty.init ();
00138                 errno_assert (rc == 0);
00139                 if (!current_out->check_write (&empty)) {
00140                     it->second.active = false;
00141                     more_out = false;
00142                     current_out = NULL;
00143                     rc = empty.close ();
00144                     errno_assert (rc == 0);
00145                     errno = EAGAIN;
00146                     return -1;
00147                 }
00148                 rc = empty.close ();
00149                 errno_assert (rc == 0);
00150             }
00151         }
00152 
00153         int rc = msg_->close ();
00154         errno_assert (rc == 0);
00155         rc = msg_->init ();
00156         errno_assert (rc == 0);
00157         return 0;
00158     }
00159 
00160     //  Check whether this is the last part of the message.
00161     more_out = msg_->flags () & msg_t::more;
00162 
00163     //  Push the message into the pipe. If there's no out pipe, just drop it.
00164     if (current_out) {
00165         bool ok = current_out->write (msg_);
00166         zmq_assert (ok);
00167         if (!more_out) {
00168             current_out->flush ();

00170         }
00171     }
00172     else {

00174         errno_assert (rc == 0);
00175     }

00184 int zmq::xrep_t::xrecv (msg_t *msg_, int flags_)
00185 {
00186     //  If there is a prefetched message, return it.
00187     if (prefetched) {
00188         int rc = msg_->move (prefetched_msg);
00189         errno_assert (rc == 0);
00190         more_in = msg_->flags () & msg_t::more;
00191         prefetched = false;
00192         return 0;
00193     }
00194 
00195     //  Deallocate old content of the message.
00196     int rc = msg_->close ();
00197     errno_assert (rc == 0);
00198 
00199     //  If we are in the middle of reading a message, just grab next part of it.
00200     if (more_in) {
00201         zmq_assert (inpipes [current_in].active);
00202         bool fetched = inpipes [current_in].pipe->read (msg_);
00203         zmq_assert (fetched);
00204         more_in = msg_->flags () & msg_t::more;
00205         if (!more_in) {
00206             current_in++;
00207             if (current_in >= inpipes.size ())
00208                 current_in = 0;
00209         }
00210         return 0;
00211     }
00212 
00213     //  Round-robin over the pipes to get the next message.
00214     for (inpipes_t::size_type count = inpipes.size (); count != 0; count--) {
00215 
00216         //  Try to fetch new message.
00217         if (inpipes [current_in].active)
00218             prefetched = inpipes [current_in].pipe->read (&prefetched_msg);
00219 
00220         //  If we have a message, create a prefix and return it to the caller.
00221         if (prefetched) {
00222             int rc = msg_->init_size (inpipes [current_in].identity.size ());
00223             errno_assert (rc == 0);
00224             memcpy (msg_->data (), inpipes [current_in].identity.data (),
00225                 msg_->size ());
00226             msg_->set_flags (msg_t::more);
00227             return 0;
00228         }
00229 
00230         //  If me don't have a message, mark the pipe as passive and
00231         //  move to next pipe.
00232         inpipes [current_in].active = false;
00233         current_in++;
00234         if (current_in >= inpipes.size ())
00235             current_in = 0;
00236     }
00237 
00238     //  No message is available. Initialise the output parameter
00239     //  to be a 0-byte message.
00240     rc = msg_->init ();
00241     errno_assert (rc == 0);
00242     errno = EAGAIN;
00243     return -1;
00244 }

00257 {

00259     if (prefetched || more_in)

00267         if (inpipes [current_in].active &&
