00037 int zmq::rep_t::xsend (msg_t *msg_, int flags_)
00038 {
00039     //  If we are in the middle of receiving a request, we cannot send reply.
00040     if (!sending_reply) {
00041         errno = EFSM;
00042         return -1;
00043     }
00044 
00045     bool more = msg_->flags () & (msg_t::more | msg_t::label) ? true : false;
00046 
00047     //  Push message to the reply pipe.
00048     int rc = xrep_t::xsend (msg_, flags_);
00049     if (rc != 0)
00050         return rc;
00051 
00052     //  If the reply is complete flip the FSM back to request receiving state.
00053     if (!more)
00054         sending_reply = false;
00055 
00056     return 0;
00057 }

00059 int zmq::rep_t::xrecv (msg_t *msg_, int flags_)
00060 {
00061     //  If we are in middle of sending a reply, we cannot receive next request.
00062     if (sending_reply) {
00063         errno = EFSM;
00064         return -1;
00065     }
00066 
00067     //  First thing to do when receiving a request is to copy all the labels
00068     //  to the reply pipe.
00069     if (request_begins) {
00070         while (true) {
00071             int rc = xrep_t::xrecv (msg_, flags_);
00072             if (rc != 0)
00073                 return rc;
00074             if (!(msg_->flags () & msg_t::label))
00075                 break;
00076 
00077             //  TODO: If the reply cannot be sent to the peer because
00078             //  od congestion, we should drop it.
00079             rc = xrep_t::xsend (msg_, flags_);
00080             zmq_assert (rc == 0);
00081         }
00082         request_begins = false;
00083     }
00084     else {
00085         int rc = xrep_t::xrecv (msg_, flags_);
00086         if (rc != 0)
00087             return rc;
00088     }
00089     zmq_assert (!(msg_->flags () & msg_t::label));
00090 
00091     //  If whole request is read, flip the FSM to reply-sending state.
00092     if (!(msg_->flags () & msg_t::more)) {
00093         sending_reply = true;
00094         request_begins = true;
00095     }
00096 
00097     return 0;
00098 }
