00105         {
00106 #if defined ZMQ_ATOMIC_PTR_WINDOWS

00111 #elif defined ZMQ_ATOMIC_PTR_X86

00114                 "lock; cmpxchg %2, %3"
00115                 : "=a" (old), "=m" (ptr)
