00052 bool zmq::decoder_t::one_byte_size_ready ()
00053 {
00054     //  First byte of size is read. If it is 0xff read 8-byte size.
00055     //  Otherwise allocate the buffer for message data and read the
00056     //  message data into it.
00057     if (*tmpbuf == 0xff)
00058         next_step (tmpbuf, 8, &decoder_t::eight_byte_size_ready);
00059     else {
00060 
00061         //  There has to be at least one byte (the flags) in the message).
00062         if (!*tmpbuf) {
00063             decoding_error ();
00064             return false;
00065         }
00066 
00067         //  in_progress is initialised at this point so in theory we should
00068         //  close it before calling zmq_msg_init_size, however, it's a 0-byte
00069         //  message and thus we can treat it as uninitialised...
00070         int rc;
00071         if (maxmsgsize >= 0 && (int64_t) (*tmpbuf - 1) > maxmsgsize) {
00072             rc = -1;
00073             errno = ENOMEM;
00074         }
00075         else
00076             rc = in_progress.init_size (*tmpbuf - 1);
00077         if (rc != 0 && errno == ENOMEM) {
00078             rc = in_progress.init ();
00079             errno_assert (rc == 0);
00080             decoding_error ();
00081             return false;
00082         }
00083         errno_assert (rc == 0);
00084 
00085         next_step (tmpbuf, 1, &decoder_t::flags_ready);
00086     }
00087     return true;
00088 }
