00053 bool zmq::decoder_t::one_byte_size_ready ()
00054 {
00055     //  First byte of size is read. If it is 0xff read 8-byte size.
00056     //  Otherwise allocate the buffer for message data and read the
00057     //  message data into it.
00058     if (*tmpbuf == 0xff)
00059         next_step (tmpbuf, 8, &decoder_t::eight_byte_size_ready);
00060     else {
00061 
00062         //  There has to be at least one byte (the flags) in the message).
00063         if (!*tmpbuf) {
00064             decoding_error ();
00065             return false;
00066         }
00067 
00068         //  in_progress is initialised at this point so in theory we should
00069         //  close it before calling zmq_msg_init_size, however, it's a 0-byte
00070         //  message and thus we can treat it as uninitialised...
00071         int rc;
00072         if (maxmsgsize >= 0 && (int64_t) (*tmpbuf - 1) > maxmsgsize) {
00073             rc = -1;
00074             errno = ENOMEM;
00075         }
00076         else
00077             rc = in_progress.init_size (*tmpbuf - 1);
00078         if (rc != 0 && errno == ENOMEM) {
00079             rc = in_progress.init ();
00080             errno_assert (rc == 0);
00081             decoding_error ();
00082             return false;
00083         }
00084         errno_assert (rc == 0);
00085 
00086         next_step (tmpbuf, 1, &decoder_t::flags_ready);
00087     }
00088     return true;
00089 }
