00069 int zmq::fq_t::recvpipe (msg_t *msg_, int flags_, pipe_t **pipe_)
00070 {
00071     //  Deallocate old content of the message.
00072     int rc = msg_->close ();
00073     errno_assert (rc == 0);
00074 
00075     //  Round-robin over the pipes to get the next message.
00076     for (pipes_t::size_type count = active; count != 0; count--) {
00077 
00078         //  Try to fetch new message. If we've already read part of the message
00079         //  subsequent part should be immediately available.
00080         bool fetched = pipes [current]->read (msg_);
00081 
00082         //  Check the atomicity of the message. If we've already received the
00083         //  first part of the message we should get the remaining parts
00084         //  without blocking.
00085         zmq_assert (!(more && !fetched));
00086 
00087         //  Note that when message is not fetched, current pipe is deactivated
00088         //  and replaced by another active pipe. Thus we don't have to increase
00089         //  the 'current' pointer.
00090         if (fetched) {
00091             if (pipe_)
00092                 *pipe_ = pipes [current];
00093             more =
00094                 msg_->flags () & (msg_t::more | msg_t::label) ? true : false;
00095             if (!more) {
00096                 current++;
00097                 if (current >= active)
00098                     current = 0;
00099             }
00100             return 0;
00101         }
00102         else {
00103             active--;
00104             pipes.swap (current, active);
00105             if (current == active)
00106                 current = 0;
00107         }
00108     }
00109 
00110     //  No message is available. Initialise the output parameter
00111     //  to be a 0-byte message.
00112     rc = msg_->init ();
00113     errno_assert (rc == 0);
00114     errno = EAGAIN;
00115     return -1;
00116 }

00118 bool zmq::fq_t::has_in ()
00119 {
00120     //  There are subsequent parts of the partly-read message available.
00121     if (more)
00122         return true;
00123 
00124     //  Note that messing with current doesn't break the fairness of fair
00125     //  queueing algorithm. If there are no messages available current will
00126     //  get back to its original value. Otherwise it'll point to the first
00127     //  pipe holding messages, skipping only pipes with no messages available.
00128     for (pipes_t::size_type count = active; count != 0; count--) {
00129         if (pipes [current]->check_read ())
00130             return true;
00131 
00132         //  Deactivate the pipe.
00133         active--;
00134         pipes.swap (current, active);
00135         if (current == active)
00136             current = 0;
00137     }
