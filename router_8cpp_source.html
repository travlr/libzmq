00048 void zmq::router_t::xattach_pipe (pipe_t *pipe_)
00049 {
00050     zmq_assert (pipe_);
00051 
00052     //  Generate a new peer ID. Take care to avoid duplicates.
00053     outpipes_t::iterator it = outpipes.lower_bound (next_peer_id);
00054     if (!outpipes.empty ()) {
00055         while (true) {
00056             if (it == outpipes.end ())
00057                 it = outpipes.begin ();
00058             if (it->first != next_peer_id)
00059                 break;
00060             ++next_peer_id;
00061             ++it;
00062         }
00063     }
00064 
00065     //  Add the pipe to the map out outbound pipes.
00066     outpipe_t outpipe = {pipe_, true};
00067     bool ok = outpipes.insert (outpipes_t::value_type (
00068         next_peer_id, outpipe)).second;
00069     zmq_assert (ok);
00070 
00071     //  Add the pipe to the list of inbound pipes.
00072     pipe_->set_pipe_id (next_peer_id);
00073     fq.attach (pipe_);
00074 
00075     //  Queue the connection command.
00076     pending_command_t cmd = {1, next_peer_id};
00077     pending_commands.push_back (cmd);
00078 
00079     //  Advance next peer ID so that if new connection is dropped shortly after
00080     //  its creation we don't accidentally get two subsequent peers with

00125 int zmq::router_t::xsend (msg_t *msg_, int flags_)
00126 {
00127     //  If this is the first part of the message it's the ID of the
00128     //  peer to send the message to.
00129     if (!more_out) {
00130         zmq_assert (!current_out);
00131 
00132         //  The first message part has to be label.
00133         if (unlikely (!(msg_->flags () & msg_t::label))) {
00134             errno = EFSM;
00135             return -1;
00136         }
00137 
00138         //  Find the pipe associated with the peer ID stored in the message.
00139         if (unlikely (msg_->size () != 4)) {
00140             errno = ECANTROUTE;
00141             return -1;
00142         }
00143         uint32_t peer_id = get_uint32 ((unsigned char*) msg_->data ());
00144         outpipes_t::iterator it = outpipes.find (peer_id);
00145         if (unlikely (it == outpipes.end ())) {
00146             errno = ECANTROUTE;
00147             return -1;
00148         }
00149 
00150         //  Check whether the pipe is available for writing.
00151         msg_t empty;
00152         int rc = empty.init ();
00153         errno_assert (rc == 0);
00154         if (!it->second.pipe->check_write (&empty)) {
00155             rc = empty.close ();
00156             errno_assert (rc == 0);
00157             it->second.active = false;
00158             errno = EAGAIN;
00159             return -1;
00160         }
00161         rc = empty.close ();
00162         errno_assert (rc == 0);
00163 
00164         //  Mark the pipe to send the message to.
00165         current_out = it->second.pipe;
00166         more_out = true;
00167         
00168         //  Clean up the message object.
00169         rc = msg_->close ();
00170         errno_assert (rc == 0);
00171         rc = msg_->init ();
00172         errno_assert (rc == 0);
00173         return 0;
00174     }
00175 
00176     //  Check whether this is the last part of the message.
00177     more_out = msg_->flags () & (msg_t::more | msg_t::label) ? true : false;
00178 
00179     //  Push the message into the pipe. If there's no out pipe, just drop it.
00180     if (current_out) {
00181         bool ok = current_out->write (msg_);
00182         zmq_assert (ok);
00183         if (!more_out) {
00184             current_out->flush ();
00185             current_out = NULL;
00186         }
00187     }
00188     else {
00189         int rc = msg_->close ();
00190         errno_assert (rc == 0);
00191     }
00192 
00193     //  Detach the message from the data buffer.
00194     int rc = msg_->init ();
00195     errno_assert (rc == 0);
00196 
00197     return 0;

00201 {
