00041 {

00128 void zmq::dist_t::distribute (msg_t *msg_, int flags_)
00129 {
00130     //  If there are no matching pipes available, simply drop the message.
00131     if (matching == 0) {
00132         int rc = msg_->close ();
00133         errno_assert (rc == 0);
00134         rc = msg_->init ();
00135         zmq_assert (rc == 0);
00136         return;
00137     }
00138 
00139     //  Add matching-1 references to the message. We already hold one reference,
00140     //  that's why -1.
00141     msg_->add_refs (matching - 1);
00142 
00143     //  Push copy of the message to each matching pipe.
00144     for (pipes_t::size_type i = 0; i < matching;) {
00145         if (!write (pipes [i], msg_))
00146             msg_->rm_refs (1);
00147         else
00148             i++;
00149     }
00150 
00151     //  Detach the original message from the data buffer. Note that we don't
00152     //  close the message. That's because we've already used all the references.
00153     int rc = msg_->init ();
00154     errno_assert (rc == 0);
00155 }
