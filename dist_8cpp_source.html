00122 void zmq::dist_t::distribute (msg_t *msg_, int flags_)
00123 {
00124     //  If there are no active pipes available, simply drop the message.
00125     if (active == 0) {
00126         int rc = msg_->close ();
00127         errno_assert (rc == 0);
00128         rc = msg_->init ();
00129         zmq_assert (rc == 0);
00130         return;
00131     }
00132 
00133     //  Add active-1 references to the message. We already hold one reference,
00134     //  that's why -1.
00135     msg_->add_refs (active - 1);
00136 
00137     //  Push copy of the message to each active pipe.
00138     for (pipes_t::size_type i = 0; i < active;) {
00139         if (!write (pipes [i], msg_))
00140             msg_->rm_refs (1);
00141         else
00142             i++;
00143     }
00144 
00145     //  Detach the original message from the data buffer. Note that we don't
00146     //  close the message. That's because we've already used all the references.
00147     int rc = msg_->init ();
00148     errno_assert (rc == 0);
00149 }
