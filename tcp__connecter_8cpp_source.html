00126 void zmq::tcp_connecter_t::start_connecting ()
00127 {
00128     //  Open the connecting socket.
00129     int rc = open ();
00130 
00131     //  Connect may succeed in synchronous manner.
00132     if (rc == 0) {
00133         handle = add_fd (s);
00134         handle_valid = true;
00135         out_event ();
00136         return;
00137     }
00138 
00139     //  Connection establishment may be dealyed. Poll for its completion.
00140     else if (rc == -1 && errno == EAGAIN) {
00141         handle = add_fd (s);
00142         handle_valid = true;
00143         set_pollout (handle);
00144         return;
00145     }
00146 
00147     //  Handle any other error condition by eventual reconnect.
00148     wait = true;
00149     add_reconnect_timer();

00158 {
00159     //  The new interval is the current interval + random value.
00160     int this_interval = current_reconnect_ivl +

00162 
00163     //  Only change the current reconnect interval  if the maximum reconnect

00166         options.reconnect_ivl_max > options.reconnect_ivl) {

00172         }

00182 int zmq::tcp_connecter_t::open ()
00183 {
00184     zmq_assert (s == retired_fd);
00185 
00186     //  Create the socket.
00187     s = socket (address.family (), SOCK_STREAM, IPPROTO_TCP);
00188 #ifdef ZMQ_HAVE_WINDOWS
00189     if (s == INVALID_SOCKET) {
00190         wsa_error_to_errno ();
00191         return -1;
00192     }
00193 #else
00194     if (s == -1)
00195         return -1;
00196 #endif
00197 
00198     //  On some systems, IPv4 mapping in IPv6 sockets is disabled by default.
00199     //  Switch it on in such cases.
00200     if (address.family () == AF_INET6)
00201         enable_ipv4_mapping (s);
00202 
00203     // Set the socket to non-blocking mode so that we get async connect().
00204     unblock_socket (s);
00205 
00206     //  Connect to the remote peer.
00207     int rc = ::connect (s, address.addr (), address.addrlen ());
00208 
00209     //  Connect was successfull immediately.
00210     if (rc == 0)
00211         return 0;
00212 
00213     //  Asynchronous connect was launched.
00214 #ifdef ZMQ_HAVE_WINDOWS
00215     if (rc == SOCKET_ERROR && (WSAGetLastError () == WSAEINPROGRESS ||
00216           WSAGetLastError () == WSAEWOULDBLOCK)) {
00217         errno = EAGAIN;
00218         return -1;
00219     }    
00220     wsa_error_to_errno ();
00221 #else
00222     if (rc == -1 && errno == EINPROGRESS) {
00223         errno = EAGAIN;
00224         return -1;
00225     }
00226 #endif
00227     return -1;

00230 zmq::fd_t zmq::tcp_connecter_t::connect ()
00231 {
00232     //  Async connect have finished. Check whether an error occured.
00233     int err = 0;
00234 #if defined ZMQ_HAVE_HPUX
00235     int len = sizeof (err);
00236 #else
00237     socklen_t len = sizeof (err);
00238 #endif
00239 
00240     int rc = getsockopt (s, SOL_SOCKET, SO_ERROR, (char*) &err, &len);
00241 
00242     //  Assert if the error was caused by 0MQ bug.
00243     //  Networking problems are OK. No need to assert.
00244 #ifdef ZMQ_HAVE_WINDOWS
00245     zmq_assert (rc == 0);
00246     if (err != 0) {
00247         if (err == WSAECONNREFUSED || err == WSAETIMEDOUT ||
00248               err == WSAECONNABORTED || err == WSAEHOSTUNREACH ||
00249               err == WSAENETUNREACH || err == WSAENETDOWN)
00250             return retired_fd;
00251         wsa_assert_no (err);
00252     }
00253 #else
00254 
00255     //  Following code should handle both Berkeley-derived socket
00256     //  implementations and Solaris.
00257     if (rc == -1)
00258         err = errno;
00259     if (err != 0) {
00260         errno = err;
00261         errno_assert (errno == ECONNREFUSED || errno == ECONNRESET ||
00262             errno == ETIMEDOUT || errno == EHOSTUNREACH ||
00263             errno == ENETUNREACH || errno == ENETDOWN);
00264         return retired_fd;
00265     }
00266 #endif
00267 
00268     //  Return the newly connected socket.
00269     fd_t result = s;
00270     s = retired_fd;
00271     return result;
