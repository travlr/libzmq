00053 int zmq::vtcp_listener_t::set_address (const char *addr_)
00054 {
00055     //  VTCP doesn't allow for binding to a specific interface. Connection
00056     //  string has to begin with *: (INADDR_ANY).
00057     if (strlen (addr_) < 2 || addr_ [0] != '*' || addr_ [1] != ':') {
00058         errno = EADDRNOTAVAIL;
00059         return -1;
00060     }
00061 
00062     //  Parse port and subport.
00063     uint16_t port;
00064     uint32_t subport;
00065     const char *delimiter = strrchr (addr_, '.');
00066     if (!delimiter) {
00067         port = 9220;
00068         subport = (uint32_t) atoi (addr_ + 2);
00069     }
00070     else {
00071         std::string port_str (addr_ + 2, delimiter - addr_ - 2);
00072         std::string subport_str (delimiter + 1);
00073         port = (uint16_t) atoi (port_str.c_str ());
00074         subport = (uint32_t) atoi (subport_str.c_str ());
00075     }
00076 
00077     //  Start listening.
00078     s = vtcp_bind (port, subport);
00079     if (s == retired_fd)
00080         return -1;
00081 
00082     return 0;
00083 }
