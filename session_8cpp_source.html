00290 void zmq::session_t::process_term (int linger_)
00291 {
00292     zmq_assert (state == active);
00293     state = pending;
00294 
00295     //  If linger is set to zero, we can terminate the session straight away
00296     //  not waiting for the pending messages to be sent.
00297     if (linger_ == 0) {
00298         proceed_with_term ();
00299         return;
00300     }
00301 
00302     //  If there's finite linger value, set up a timer.
00303     if (linger_ > 0) {
00304        zmq_assert (!has_linger_timer);
00305        add_timer (linger_, linger_timer_id);
00306        has_linger_timer = true;
00307     }
00308 
00309     //  If there's no engine and there's only delimiter in the pipe it wouldn't
00310     //  be ever read. Thus we check for it explicitly.
00311     if (in_pipe)
00312         in_pipe->check_read ();
00313 
00314     //  If there's no in pipe there are no pending messages to send.
00315     //  We can proceed with the shutdown straight away. Also, if there is
00316     //  inbound pipe, but the delimiter was already processed, we can
00317     //  terminate immediately. Alternatively, if the derived session type have
00318     //  called 'terminate' we'll finish straight away.
00319     if (!options.requires_out || delimiter_processed || force_terminate ||
00320         (!options.immediate_connect && !in_pipe))
00321         proceed_with_term ();
