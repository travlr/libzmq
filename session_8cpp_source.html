00096 void zmq::session_t::clean_pipes ()
00097 {
00098     if (pipe) {
00099 
00100         //  Get rid of half-processed messages in the out pipe. Flush any
00101         //  unflushed messages upstream.
00102         pipe->rollback ();
00103         pipe->flush ();
00104 
00105         //  Remove any half-read message from the in pipe.
00106         while (incomplete_in) {
00107             msg_t msg;
00108             int rc = msg.init ();
00109             errno_assert (rc == 0);
00110             if (!read (&msg)) {
00111                 zmq_assert (!incomplete_in);
00112                 break;
00113             }
00114             rc = msg.close ();
00115             errno_assert (rc == 0);
00116         }
00117     }

00162 void zmq::session_t::process_attach (i_engine *engine_,
00163     const blob_t &peer_identity_)
00164 {
00165     //  If some other object (e.g. init) notifies us that the connection failed
00166     //  without creating an engine we need to start the reconnection process.
00167     if (!engine_) {
00168         zmq_assert (!engine);
00169         detached ();
00170         return;
00171     }
00172 
00173     //  Trigger the notfication event about the attachment.
00174     if (!attached (peer_identity_)) {
00175         delete engine_;
00176         return;
00177     }
00178 
00179     //  Create the pipe if it does not exist yet.
00180     if (!pipe && !is_terminating ()) {
00181         object_t *parents [2] = {this, socket};
00182         pipe_t *pipes [2] = {NULL, NULL};
00183         int hwms [2] = {options.rcvhwm, options.sndhwm};
00184         bool delays [2] = {options.delay_on_close, options.delay_on_disconnect};
00185         int rc = pipepair (parents, pipes, hwms, delays);
00186         errno_assert (rc == 0);
00187 
00188         //  Plug the local end of the pipe.
00189         pipes [0]->set_event_sink (this);
00190 
00191         //  Remember the local end of the pipe.
00192         zmq_assert (!pipe);
00193         pipe = pipes [0];
00194 
00195         //  Ask socket to plug into the remote end of the pipe.
00196         send_bind (socket, pipes [1], peer_identity_);
00197     }
00198 
00199     //  Plug in the engine.
00200     zmq_assert (!engine);
00201     engine = engine_;
00202     engine->plug (io_thread, this);
00203 }

00221 void zmq::session_t::process_term (int linger_)
00222 {
00223     zmq_assert (!pending);
00224 
00225     //  If the termination of the pipe happens before the term command is
00226     //  delivered there's nothing much to do. We can proceed with the
00227     //  stadard termination immediately.
00228     if (!pipe) {
00229         proceed_with_term ();
00230         return;
00231     }
00232 
00233     pending = true;
00234 
00235     //  If there's finite linger value, delay the termination.
00236     //  If linger is infinite (negative) we don't even have to set
00237     //  the timer.
00238     if (linger_ > 0) {
00239         zmq_assert (!has_linger_timer);
00240         add_timer (linger_, linger_timer_id);
00241         has_linger_timer = true;
00242     }
00243 
00244     //  Start pipe termination process. Delay the termination till all messages
00245     //  are processed in case the linger time is non-zero.
00246     pipe->terminate (linger_ != 0);
00247 
00248     //  TODO: Should this go into pipe_t::terminate ?
00249     //  In case there's no engine and there's only delimiter in the
00250     //  pipe it wouldn't be ever read. Thus we check for it explicitly.
00251     pipe->check_read ();
