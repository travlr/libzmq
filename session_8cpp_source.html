00175 void zmq::session_t::process_attach (i_engine *engine_)
00176 {
00177     //  If some other object (e.g. init) notifies us that the connection failed
00178     //  without creating an engine we need to start the reconnection process.
00179     if (!engine_) {
00180         zmq_assert (!engine);
00181         detached ();
00182         return;
00183     }
00184 
00185     //  Create the pipe if it does not exist yet.
00186     if (!pipe && !is_terminating ()) {
00187         object_t *parents [2] = {this, socket};
00188         pipe_t *pipes [2] = {NULL, NULL};
00189         int hwms [2] = {options.rcvhwm, options.sndhwm};
00190         bool delays [2] = {options.delay_on_close, options.delay_on_disconnect};
00191         int rc = pipepair (parents, pipes, hwms, delays);
00192         errno_assert (rc == 0);
00193 
00194         //  Plug the local end of the pipe.
00195         pipes [0]->set_event_sink (this);
00196 
00197         //  Remember the local end of the pipe.
00198         zmq_assert (!pipe);
00199         pipe = pipes [0];
00200 
00201         //  Ask socket to plug into the remote end of the pipe.
00202         send_bind (socket, pipes [1]);
00203     }
00204 
00205     //  Plug in the engine.
00206     zmq_assert (!engine);
00207     engine = engine_;

00209 }

00228 {
00229     zmq_assert (!pending);
00230 
00231     //  If the termination of the pipe happens before the term command is
00232     //  delivered there's nothing much to do. We can proceed with the
00233     //  stadard termination immediately.
00234     if (!pipe) {

00246         add_timer (linger_, linger_timer_id);
00247         has_linger_timer = true;

00249 
00250     //  Start pipe termination process. Delay the termination till all messages
00251     //  are processed in case the linger time is non-zero.
00252     pipe->terminate (linger_ != 0);

00254     //  TODO: Should this go into pipe_t::terminate ?
