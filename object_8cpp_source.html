00058 void zmq::object_t::process_command (command_t &cmd_)
00059 {
00060     switch (cmd_.type) {
00061 
00062     case command_t::activate_reader:
00063         process_activate_reader ();
00064         break;
00065 
00066     case command_t::activate_writer:
00067         process_activate_writer (cmd_.args.activate_writer.msgs_read);
00068         break;
00069 
00070     case command_t::stop:
00071         process_stop ();
00072         break;
00073 
00074     case command_t::plug:
00075         process_plug ();
00076         process_seqnum ();
00077         return;
00078 
00079     case command_t::own:
00080         process_own (cmd_.args.own.object);
00081         process_seqnum ();
00082         break;
00083 
00084     case command_t::attach:
00085         process_attach (cmd_.args.attach.engine,
00086             cmd_.args.attach.peer_identity ?
00087             blob_t (cmd_.args.attach.peer_identity,
00088             cmd_.args.attach.peer_identity_size) : blob_t ());
00089         process_seqnum ();
00090         break;
00091 
00092     case command_t::bind:
00093         process_bind (cmd_.args.bind.in_pipe, cmd_.args.bind.out_pipe,
00094             cmd_.args.bind.peer_identity ? blob_t (cmd_.args.bind.peer_identity,
00095             cmd_.args.bind.peer_identity_size) : blob_t ());
00096         process_seqnum ();
00097         break;
00098 
00099     case command_t::pipe_term:
00100         process_pipe_term ();
00101         return;
00102 
00103     case command_t::pipe_term_ack:
00104         process_pipe_term_ack ();
00105         break;
00106 
00107     case command_t::term_req:
00108         process_term_req (cmd_.args.term_req.object);
00109         break;
00110     
00111     case command_t::term:
00112         process_term (cmd_.args.term.linger);
00113         break;
00114 
00115     case command_t::term_ack:
00116         process_term_ack ();
00117         break;
00118 
00119     case command_t::reap:
00120         process_reap (cmd_.args.reap.socket);
00121         break;
00122 
00123     case command_t::reaped:
00124         process_reaped ();
00125         break;
00126 
00127     default:
00128         zmq_assert (false);
00129     }
00130 
00131     //  The assumption here is that each command is processed once only,
00132     //  so deallocating it after processing is all right.
00133     deallocate_command (&cmd_);
