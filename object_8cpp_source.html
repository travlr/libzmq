00059 {
00060     switch (cmd_.type) {
00061 
00062     case command_t::activate_read:
00063         process_activate_read ();
00064         break;
00065 
00066     case command_t::activate_write:
00067         process_activate_write (cmd_.args.activate_write.msgs_read);
00068         break;
00069 
00070     case command_t::stop:
00071         process_stop ();
00072         break;
00073 
00074     case command_t::plug:
00075         process_plug ();
00076         process_seqnum ();
00077         break;
00078 
00079     case command_t::own:
00080         process_own (cmd_.args.own.object);
00081         process_seqnum ();
00082         break;
00083 
00084     case command_t::attach:

00086             cmd_.args.attach.peer_identity ?
00087             blob_t (cmd_.args.attach.peer_identity,
00088             cmd_.args.attach.peer_identity_size) : blob_t ());

00090         break;

00097         break;

00113         break;

00118 
00119     case command_t::term_ack:
