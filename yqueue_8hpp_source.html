00033     //  yqueue is an efficient queue implementation. The main goal is
00034     //  to minimise number of allocations/deallocations needed. Thus yqueue
00035     //  allocates/deallocates elements in batches of N.
00036     //
00037     //  yqueue allows one thread to use push/back function and another one 
00038     //  to use pop/front functions. However, user must ensure that there's no
00039     //  pop on the empty queue and that both threads don't access the same
00040     //  element in unsynchronised manner.
00041     //
00042     //  T is the type of the object in the queue.
00043     //  N is granularity of the queue (how many pushes have to be done till
00044     //  actual memory allocation is required).
00045 
00046     template <typename T, int N> class yqueue_t
00047     {
00048     public:
00049 
00050         //  Create the queue.
00051         inline yqueue_t ()
00052         {
00053              begin_chunk = (chunk_t*) malloc (sizeof (chunk_t));
00054              alloc_assert (begin_chunk);
00055              begin_pos = 0;
00056              back_chunk = NULL;
00057              back_pos = 0;
00058              end_chunk = begin_chunk;
00059              end_pos = 0;
00060         }
00061 
00062         //  Destroy the queue.
00063         inline ~yqueue_t ()
00064         {
00065             while (true) {
00066                 if (begin_chunk == end_chunk) {
00067                     free (begin_chunk);
00068                     break;
00069                 } 
00070                 chunk_t *o = begin_chunk;
00071                 begin_chunk = begin_chunk->next;
00072                 free (o);
00073             }
00074 
00075             chunk_t *sc = spare_chunk.xchg (NULL);
00076             if (sc)
00077                 free (sc);
00078         }
00079 
00080         //  Returns reference to the front element of the queue.
00081         //  If the queue is empty, behaviour is undefined.
00082         inline T &front ()
00083         {
00084              return begin_chunk->values [begin_pos];
00085         }
00086 
00087         //  Returns reference to the back element of the queue.
00088         //  If the queue is empty, behaviour is undefined.
00089         inline T &back ()
00090         {
00091             return back_chunk->values [back_pos];
00092         }
00093 
00094         //  Adds an element to the back end of the queue.
00095         inline void push ()
00096         {
00097             back_chunk = end_chunk;
00098             back_pos = end_pos;
00099 
00100             if (++end_pos != N)
00101                 return;
00102 
00103             chunk_t *sc = spare_chunk.xchg (NULL);
00104             if (sc) {
00105                 end_chunk->next = sc;
00106                 sc->prev = end_chunk;
00107             } else {
00108                 end_chunk->next = (chunk_t*) malloc (sizeof (chunk_t));
00109                 alloc_assert (end_chunk->next);
00110                 end_chunk->next->prev = end_chunk;
00111             }
00112             end_chunk = end_chunk->next;
00113             end_pos = 0;
00114         }
00115 
00116         //  Removes element from the back end of the queue. In other words
00117         //  it rollbacks last push to the queue. Take care: Caller is
00118         //  responsible for destroying the object being unpushed.
00119         //  The caller must also guarantee that the queue isn't empty when
00120         //  unpush is called. It cannot be done automatically as the read
00121         //  side of the queue can be managed by different, completely
00122         //  unsynchronised thread.
00123         inline void unpush ()
00124         {
00125             //  First, move 'back' one position backwards.
00126             if (back_pos)
00127                 --back_pos;
00128             else {
00129                 back_pos = N - 1;
00130                 back_chunk = back_chunk->prev;
00131             }
00132 
00133             //  Now, move 'end' position backwards. Note that obsolete end chunk
00134             //  is not used as a spare chunk. The analysis shows that doing so
00135             //  would require free and atomic operation per chunk deallocated
00136             //  instead of a simple free.
00137             if (end_pos)
00138                 --end_pos;
00139             else {
00140                 end_pos = N - 1;
00141                 end_chunk = end_chunk->prev;
00142                 free (end_chunk->next);
00143                 end_chunk->next = NULL;
00144             }
00145         }
00146 
00147         //  Removes an element from the front end of the queue.
00148         inline void pop ()
00149         {
00150             if (++ begin_pos == N) {
00151                 chunk_t *o = begin_chunk;
00152                 begin_chunk = begin_chunk->next;
00153                 begin_chunk->prev = NULL;
00154                 begin_pos = 0;
00155 
00156                 //  'o' has been more recently used than spare_chunk,
00157                 //  so for cache reasons we'll get rid of the spare and
00158                 //  use 'o' as the spare.
00159                 chunk_t *cs = spare_chunk.xchg (o);
00160                 if (cs)
00161                     free (cs);
00162             }
00163         }
00164

00167         //  Individual memory chunk to hold N elements.
00168         struct chunk_t
00169         {
00170              T values [N];
00171              chunk_t *prev;
00172              chunk_t *next;
00173         };
00174 
00175         //  Back position may point to invalid memory if the queue is empty,
00176         //  while begin & end positions are always valid. Begin position is
00177         //  accessed exclusively be queue reader (front/pop), while back and
00178         //  end positions are accessed exclusively by queue writer (back/push).
00179         chunk_t *begin_chunk;
00180         int begin_pos;
00181         chunk_t *back_chunk;
00182         int back_pos;
00183         chunk_t *end_chunk;

00186         //  People are likely to produce and consume at similar rates.  In
00187         //  this scenario holding onto the most recently freed chunk saves
