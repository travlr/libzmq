00151 bool zmq::trie_t::check (unsigned char *data_, size_t size_)
00152 {
00153     //  This function is on critical path. It deliberately doesn't use
00154     //  recursion to get a bit better performance.
00155     trie_t *current = this;
00156     while (true) {
00157 
00158         //  We've found a corresponding subscription!
00159         if (current->refcnt)
00160             return true;
00161 
00162         //  We've checked all the data and haven't found matching subscription.
00163         if (!size_)
00164             return false;
00165 
00166         //  If there's no corresponding slot for the first character
00167         //  of the prefix, the message does not match.
00168         unsigned char c = *data_;
00169         if (c < current->min || c >= current->min + current->count)
00170             return false;
00171 
00172         //  Move to the next character.
00173         if (current->count == 1)
00174             current = current->next.node;
00175         else {
00176             current = current->next.table [c - current->min];
00177             if (!current)
00178                 return false;
00179         }
00180         data_++;
00181         size_--;
00182     }
00183 }
