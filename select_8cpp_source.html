00084 void zmq::select_t::rm_fd (handle_t handle_)
00085 {
00086     //  Mark the descriptor as retired.
00087     fd_set_t::iterator it;
00088     for (it = fds.begin (); it != fds.end (); ++it)
00089         if (it->fd == handle_)
00090             break;
00091     zmq_assert (it != fds.end ());
00092     it->fd = retired_fd;
00093     retired = true;
00094 
00095     //  Stop polling on the descriptor.
00096     FD_CLR (handle_, &source_set_in);
00097     FD_CLR (handle_, &source_set_out);
00098     FD_CLR (handle_, &source_set_err);
00099 
00100     //  Discard all events generated on this file descriptor.
00101     FD_CLR (handle_, &readfds);
00102     FD_CLR (handle_, &writefds);
00103     FD_CLR (handle_, &exceptfds);
00104 
00105     //  Adjust the maxfd attribute if we have removed the
00106     //  highest-numbered file descriptor.
00107     if (handle_ == maxfd) {
00108         maxfd = retired_fd;
00109         for (fd_set_t::iterator it = fds.begin (); it != fds.end (); ++it)
00110             if (it->fd > maxfd)
00111                 maxfd = it->fd;
00112     }
00113 
00114     //  Decrease the load metric of the thread.
00115     adjust_load (-1);

00148 void zmq::select_t::loop ()
00149 {
00150     while (!stopping) {
00151 
00152         //  Execute any due timers.
00153         int timeout = (int) execute_timers ();
00154 
00155         //  Intialise the pollsets.
00156         memcpy (&readfds, &source_set_in, sizeof source_set_in);
00157         memcpy (&writefds, &source_set_out, sizeof source_set_out);
00158         memcpy (&exceptfds, &source_set_err, sizeof source_set_err);
00159 
00160         //  Wait for events.
00161         struct timeval tv = {(long) (timeout / 1000),
00162             (long) (timeout % 1000 * 1000)};
00163 #ifdef ZMQ_HAVE_WINDOWS
00164         int rc = select (0, &readfds, &writefds, &exceptfds,
00165             timeout ? &tv : NULL);
00166         wsa_assert (rc != SOCKET_ERROR);
00167 #else
00168         int rc = select (maxfd + 1, &readfds, &writefds, &exceptfds,
00169             timeout ? &tv : NULL);
00170         if (rc == -1 && errno == EINTR)
00171             continue;
00172         errno_assert (rc != -1);
00173 #endif
00174 
00175         //  If there are no events (i.e. it's a timeout) there's no point
00176         //  in checking the pollset.
00177         if (rc == 0)
00178             continue;
00179 
00180         for (fd_set_t::size_type i = 0; i < fds.size (); i ++) {
00181             if (fds [i].fd == retired_fd)
00182                 continue;
00183             if (FD_ISSET (fds [i].fd, &exceptfds))
00184                 fds [i].events->in_event ();
00185             if (fds [i].fd == retired_fd)
00186                 continue;
00187             if (FD_ISSET (fds [i].fd, &writefds))
00188                 fds [i].events->out_event ();
00189             if (fds [i].fd == retired_fd)
00190                 continue;
00191             if (FD_ISSET (fds [i].fd, &readfds))
00192                 fds [i].events->in_event ();
00193         }
00194 
00195         //  Destroy retired event sources.
00196         if (retired) {
00197             fds.erase (std::remove_if (fds.begin (), fds.end (),
00198                 zmq::select_t::is_retired_fd), fds.end ());
00199             retired = false;
00200         }
00201     }
00202 }
