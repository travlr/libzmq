00082 void zmq::select_t::rm_fd (handle_t handle_)
00083 {
00084     //  Mark the descriptor as retired.
00085     fd_set_t::iterator it;
00086     for (it = fds.begin (); it != fds.end (); ++it)
00087         if (it->fd == handle_)
00088             break;
00089     zmq_assert (it != fds.end ());
00090     it->fd = retired_fd;

00092 
00093     //  Stop polling on the descriptor.
00094     FD_CLR (handle_, &source_set_in);
00095     FD_CLR (handle_, &source_set_out);
00096     FD_CLR (handle_, &source_set_err);

00098     //  Discard all events generated on this file descriptor.
00099     FD_CLR (handle_, &readfds);
00100     FD_CLR (handle_, &writefds);
00101     FD_CLR (handle_, &exceptfds);
00102 
00103     //  Adjust the maxfd attribute if we have removed the
00104     //  highest-numbered file descriptor.
00105     if (handle_ == maxfd) {
00106         maxfd = retired_fd;

00108             if (it->fd > maxfd)

00146 void zmq::select_t::loop ()
00147 {
00148     while (!stopping) {
00149 
00150         //  Execute any due timers.
00151         int timeout = (int) execute_timers ();
00152 
00153         //  Intialise the pollsets.
00154         memcpy (&readfds, &source_set_in, sizeof source_set_in);
00155         memcpy (&writefds, &source_set_out, sizeof source_set_out);
00156         memcpy (&exceptfds, &source_set_err, sizeof source_set_err);
00157 
00158         //  Wait for events.
00159         struct timeval tv = {(long) (timeout / 1000),
00160             (long) (timeout % 1000 * 1000)};
00161 #ifdef ZMQ_HAVE_WINDOWS
00162         int rc = select (0, &readfds, &writefds, &exceptfds,
00163             timeout ? &tv : NULL);
00164         wsa_assert (rc != SOCKET_ERROR);
00165 #else
00166         int rc = select (maxfd + 1, &readfds, &writefds, &exceptfds,
00167             timeout ? &tv : NULL);
00168         if (rc == -1 && errno == EINTR)
00169             continue;
00170         errno_assert (rc != -1);
00171 #endif
00172 
00173         //  If there are no events (i.e. it's a timeout) there's no point
00174         //  in checking the pollset.
00175         if (rc == 0)
00176             continue;
00177 
00178         for (fd_set_t::size_type i = 0; i < fds.size (); i ++) {
00179             if (fds [i].fd == retired_fd)
00180                 continue;
00181             if (FD_ISSET (fds [i].fd, &exceptfds))
00182                 fds [i].events->in_event ();
00183             if (fds [i].fd == retired_fd)
00184                 continue;
00185             if (FD_ISSET (fds [i].fd, &writefds))
00186                 fds [i].events->out_event ();
00187             if (fds [i].fd == retired_fd)
00188                 continue;
00189             if (FD_ISSET (fds [i].fd, &readfds))
00190                 fds [i].events->in_event ();
00191         }
00192 
00193         //  Destroy retired event sources.
00194         if (retired) {
00195             fds.erase (std::remove_if (fds.begin (), fds.end (),
00196                 zmq::select_t::is_retired_fd), fds.end ());
00197             retired = false;
00198         }
00199     }
00200 }
