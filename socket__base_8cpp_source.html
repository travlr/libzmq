00508 {
00509     //  Check whether the library haven't been shut down yet.
00510     if (unlikely (ctx_terminated)) {
00511         errno = ETERM;
00512         return -1;
00513     }
00514 
00515     //  Check whether message passed to the function is valid.
00516     if (unlikely (!msg_->check ())) {
00517         errno = EFAULT;
00518         return -1;
00519     }
00520 
00521     //  Process pending commands, if any.
00522     int rc = process_commands (0, true);
00523     if (unlikely (rc != 0))
00524         return -1;

00526     //  At this point we impose the flags on the message.

00529     if (flags_ & ZMQ_SNDMORE)
00530         msg_->set_flags (msg_t::more);
00531     if (flags_ & ZMQ_SNDCMD)
00532         msg_->set_flags (msg_t::command);
00533 
00534     //  Try to send the message.
00535     rc = xsend (msg_, flags_);

00537         return 0;

00539         return -1;
00540 
00541     //  In case of non-blocking send we'll simply propagate
00542     //  the error - including EAGAIN - up the stack.
00543     if (flags_ & ZMQ_DONTWAIT || options.sndtimeo == 0)
00544         return -1;
00545 
00546     //  Compute the time when the timeout should occur.
00547     //  If the timeout is infite, don't care.

00551

00556         if (unlikely (process_commands (timeout, false) != 0))
00557             return -1;

00559         if (rc == 0)
00560             break;
00561         if (unlikely (errno != EAGAIN))
00562             return -1;
00563         if (timeout > 0) {
00564             timeout = (int) (end - clock.now_ms ());
00565             if (timeout <= 0) {

00569         }
00570     }

00575 {

00578         errno = ETERM;
00579         return -1;
00580     }
00581 
00582     //  Check whether message passed to the function is valid.
00583     if (unlikely (!msg_->check ())) {
00584         errno = EFAULT;
00585         return -1;
00586     }

00591         return -1;

00594     //  incoming commands. This happens only if we are not polling altogether
00595     //  because there are messages available all the time. If poll occurs,
00596     //  ticks is set to zero and thus we avoid this code.
00597     //
00598     //  Note that 'recv' uses different command throttling algorithm (the one
00599     //  described above) from the one used by 'send'. This is because counting
00600     //  ticks is more efficient than doing RDTSC all the time.
00601     if (++ticks == inbound_poll_rate) {
00602         if (unlikely (process_commands (0, false) != 0))
00603             return -1;

00605     }
00606

00608     if (rc == 0) {
00609         extract_flags (msg_);
00610         return 0;
00611     }
00612 
00613     //  If the message cannot be fetched immediately, there are two scenarios.
00614     //  For non-blocking recv, commands are processed in case there's an
00615     //  activate_reader command already waiting int a command pipe.
00616     //  If it's not, return EAGAIN.
00617     if (flags_ & ZMQ_DONTWAIT || options.rcvtimeo == 0) {
00618         if (unlikely (process_commands (0, false) != 0))
00619             return -1;
00620         ticks = 0;
00621 
00622         rc = xrecv (msg_, flags_);
00623         if (rc < 0)
00624             return rc;
00625         extract_flags (msg_);
00626         return 0;
00627     }

00645         }
00646         if (unlikely (errno != EAGAIN))

00654             }
00655         }

00695 int zmq::socket_base_t::process_commands (int timeout_, bool throttle_)
00696 {
00697     int rc;
00698     command_t cmd;
00699     if (timeout_ != 0) {
00700 
00701         //  If we are asked to wait, simply ask mailbox to wait.
00702         rc = mailbox.recv (&cmd, timeout_);
00703     }
00704     else {
00705 
00706         //  If we are asked not to wait, check whether we haven't processed
00707         //  commands recently, so that we can throttle the new commands.
00708 
00709         //  Get the CPU's tick counter. If 0, the counter is not available.
00710         uint64_t tsc = zmq::clock_t::rdtsc ();
00711 
00712         //  Optimised version of command processing - it doesn't have to check
00713         //  for incoming commands each time. It does so only if certain time
00714         //  elapsed since last command processing. Command delay varies
00715         //  depending on CPU speed: It's ~1ms on 3GHz CPU, ~2ms on 1.5GHz CPU
00716         //  etc. The optimisation makes sense only on platforms where getting
00717         //  a timestamp is a very cheap operation (tens of nanoseconds).
00718         if (tsc && throttle_) {
00719 
00720             //  Check whether TSC haven't jumped backwards (in case of migration
00721             //  between CPU cores) and whether certain time have elapsed since
00722             //  last command processing. If it didn't do nothing.
00723             if (tsc >= last_tsc && tsc - last_tsc <= max_command_delay)
00724                 return 0;
00725             last_tsc = tsc;
00726         }
00727 
00728         //  Check whether there are any commands pending for this thread.
00729         rc = mailbox.recv (&cmd, 0);
00730     }
00731 
00732     //  Process all the commands available at the moment.
00733     while (true) {
00734         if (rc == -1 && errno == EAGAIN)
00735             break;
00736         if (rc == -1 && errno == EINTR)
00737             return -1;
00738         errno_assert (rc == 0);
00739         cmd.destination->process_command (cmd);
00740         rc = mailbox.recv (&cmd, 0);
00741      }
00742 
00743     if (ctx_terminated) {
00744         errno = ETERM;
00745         return -1;
00746     }
00747 
00748     return 0;
00749 }
