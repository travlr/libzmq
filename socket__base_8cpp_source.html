00512 {
00513     //  Check whether the library haven't been shut down yet.
00514     if (unlikely (ctx_terminated)) {
00515         errno = ETERM;
00516         return -1;
00517     }
00518 
00519     //  Check whether message passed to the function is valid.
00520     if (unlikely (!msg_->check ())) {
00521         errno = EFAULT;
00522         return -1;
00523     }
00524 
00525     //  Get the message.
00526     int rc = xrecv (msg_, flags_);

00528 
00529     //  Once every inbound_poll_rate messages check for signals and process
00530     //  incoming commands. This happens only if we are not polling altogether
00531     //  because there are messages available all the time. If poll occurs,
00532     //  ticks is set to zero and thus we avoid this code.
00533     //
00534     //  Note that 'recv' uses different command throttling algorithm (the one
00535     //  described above) from the one used by 'send'. This is because counting
00536     //  ticks is more efficient than doing RDTSC all the time.
00537     if (++ticks == inbound_poll_rate) {
00538         if (unlikely (process_commands (false, false) != 0))
00539             return -1;
00540         ticks = 0;
00541     }
00542 
00543     //  If we have the message, return immediately.
00544     if (rc == 0) {
00545         rcvmore = msg_->flags () & msg_t::more;
00546         if (rcvmore)
00547             msg_->reset_flags (msg_t::more);
00548         return 0;
00549     }

00552     errno = err;
00553 
00554     //  If the message cannot be fetched immediately, there are two scenarios.
00555     //  For non-blocking recv, commands are processed in case there's an
00556     //  activate_reader command already waiting int a command pipe.
00557     //  If it's not, return EAGAIN.
00558     if (flags_ & ZMQ_DONTWAIT) {
00559         if (errno != EAGAIN)

00561         if (unlikely (process_commands (false, false) != 0))
00562             return -1;

00579             return -1;

00581             return -1;

00671         uint64_t tsc = zmq::clock_t::rdtsc ();
00672 
00673         //  Optimised version of command processing - it doesn't have to check
00674         //  for incoming commands each time. It does so only if certain time
00675         //  elapsed since last command processing. Command delay varies
00676         //  depending on CPU speed: It's ~1ms on 3GHz CPU, ~2ms on 1.5GHz CPU
00677         //  etc. The optimisation makes sense only on platforms where getting
00678         //  a timestamp is a very cheap operation (tens of nanoseconds).
00679         if (tsc && throttle_) {
00680 
00681             //  Check whether TSC haven't jumped backwards (in case of migration
00682             //  between CPU cores) and whether certain time have elapsed since
00683             //  last command processing. If it didn't do nothing.
00684             if (tsc >= last_tsc && tsc - last_tsc <= max_command_delay)
00685                 return 0;
00686             last_tsc = tsc;
00687         }
00688 
00689         //  Check whether there are any commands pending for this thread.
00690         rc = mailbox.recv (&cmd, false);
00691     }
00692 
00693     //  Process all the commands available at the moment.
00694     while (true) {
00695         if (rc == -1 && errno == EAGAIN)
00696             break;
00697         if (rc == -1 && errno == EINTR)
00698             return -1;
00699         errno_assert (rc == 0);
00700         cmd.destination->process_command (cmd);
00701         rc = mailbox.recv (&cmd, false);
00702      }
00703 
00704     if (ctx_terminated) {
00705         errno = ETERM;
00706         return -1;
00707     }
00708 
00709     return 0;
00710 }
