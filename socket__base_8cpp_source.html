00483 {
00484     //  Check whether the library haven't been shut down yet.
00485     if (unlikely (ctx_terminated)) {
00486         errno = ETERM;
00487         return -1;
00488     }
00489 
00490     //  Check whether message passed to the function is valid.
00491     if (unlikely (!msg_->check ())) {
00492         errno = EFAULT;
00493         return -1;
00494     }
00495 
00496     //  Process pending commands, if any.
00497     int rc = process_commands (0, true);
00498     if (unlikely (rc != 0))
00499         return -1;
00500

00503         msg_->set_flags (msg_t::label);
00504     if (flags_ & ZMQ_SNDMORE)
00505         msg_->set_flags (msg_t::more);
00506 
00507     //  Try to send the message.
00508     rc = xsend (msg_, flags_);
00509     if (rc == 0)
00510         return 0;
00511     if (unlikely (errno != EAGAIN))
00512         return -1;
00513 
00514     //  In case of non-blocking send we'll simply propagate
00515     //  the error - including EAGAIN - up the stack.
00516     if (flags_ & ZMQ_DONTWAIT || options.sndtimeo == 0)
00517         return -1;
00518 
00519     //  Compute the time when the timeout should occur.
00520     //  If the timeout is infite, don't care. 
00521     clock_t clock ;
00522     int timeout = options.sndtimeo;
00523     uint64_t end = timeout < 0 ? 0 : (clock.now_ms () + timeout);
00524 
00525     //  Oops, we couldn't send the message. Wait for the next
00526     //  command, process it and try to send the message again.
00527     //  If timeout is reached in the meantime, return EAGAIN.
00528     while (true) {
00529         if (unlikely (process_commands (timeout, false) != 0))
00530             return -1;
00531         rc = xsend (msg_, flags_);
00532         if (rc == 0)
00533             break;
00534         if (unlikely (errno != EAGAIN))
00535             return -1;

00552         return -1;
00553     }

00555     //  Check whether message passed to the function is valid.

00558         return -1;

00562     int rc = xrecv (msg_, flags_);
00563     if (unlikely (rc != 0 && errno != EAGAIN))

00567     //  incoming commands. This happens only if we are not polling altogether
00568     //  because there are messages available all the time. If poll occurs,
00569     //  ticks is set to zero and thus we avoid this code.
00570     //
00571     //  Note that 'recv' uses different command throttling algorithm (the one
00572     //  described above) from the one used by 'send'. This is because counting
00573     //  ticks is more efficient than doing RDTSC all the time.
00574     if (++ticks == inbound_poll_rate) {
00575         if (unlikely (process_commands (0, false) != 0))
00576             return -1;

00578     }
00579 
00580     //  If we have the message, return immediately.
00581     if (rc == 0) {
00582         rcvlabel = msg_->flags () & msg_t::label;
00583         if (rcvlabel)
00584             msg_->reset_flags (msg_t::label);
00585         rcvmore = msg_->flags () & msg_t::more ? true : false;
00586         if (rcvmore)
00587             msg_->reset_flags (msg_t::more);
00588         return 0;
00589     }

00592     //  For non-blocking recv, commands are processed in case there's an
00593     //  activate_reader command already waiting int a command pipe.

00595     if (flags_ & ZMQ_DONTWAIT || options.rcvtimeo == 0) {
00596         if (unlikely (process_commands (0, false) != 0))
00597             return -1;
00598         ticks = 0;

00602             return rc;

00606         rcvmore = msg_->flags () & msg_t::more ? true : false;
00607         if (rcvmore)
00608             msg_->reset_flags (msg_t::more);

00610     }
00611

00722 int zmq::socket_base_t::process_commands (int timeout_, bool throttle_)
00723 {
00724     int rc;
00725     command_t cmd;
00726     if (timeout_ != 0) {
00727 
00728         //  If we are asked to wait, simply ask mailbox to wait.
00729         rc = mailbox.recv (&cmd, timeout_);
00730     }
00731     else {
00732 
00733         //  If we are asked not to wait, check whether we haven't processed
00734         //  commands recently, so that we can throttle the new commands.
00735 
00736         //  Get the CPU's tick counter. If 0, the counter is not available.
00737         uint64_t tsc = zmq::clock_t::rdtsc ();
00738 
00739         //  Optimised version of command processing - it doesn't have to check
00740         //  for incoming commands each time. It does so only if certain time
00741         //  elapsed since last command processing. Command delay varies
00742         //  depending on CPU speed: It's ~1ms on 3GHz CPU, ~2ms on 1.5GHz CPU
00743         //  etc. The optimisation makes sense only on platforms where getting
00744         //  a timestamp is a very cheap operation (tens of nanoseconds).
00745         if (tsc && throttle_) {
00746 
00747             //  Check whether TSC haven't jumped backwards (in case of migration
00748             //  between CPU cores) and whether certain time have elapsed since
00749             //  last command processing. If it didn't do nothing.
00750             if (tsc >= last_tsc && tsc - last_tsc <= max_command_delay)
00751                 return 0;
00752             last_tsc = tsc;
00753         }
00754 
00755         //  Check whether there are any commands pending for this thread.
00756         rc = mailbox.recv (&cmd, 0);
00757     }
00758 
00759     //  Process all the commands available at the moment.
00760     while (true) {
00761         if (rc == -1 && errno == EAGAIN)
00762             break;
00763         if (rc == -1 && errno == EINTR)
00764             return -1;
00765         errno_assert (rc == 0);
00766         cmd.destination->process_command (cmd);
00767         rc = mailbox.recv (&cmd, 0);
00768      }
00769 
00770     if (ctx_terminated) {
00771         errno = ETERM;
00772         return -1;
00773     }
00774 
00775     return 0;
00776 }
