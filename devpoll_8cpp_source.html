00137 void zmq::devpoll_t::loop ()
00138 {
00139     while (!stopping) {
00140 
00141         struct pollfd ev_buf [max_io_events];
00142         struct dvpoll poll_req;
00143 
00144         for (pending_list_t::size_type i = 0; i < pending_list.size (); i ++)
00145             fd_table [pending_list [i]].accepted = true;
00146         pending_list.clear ();
00147 
00148         //  Execute any due timers.
00149         int timeout = (int) execute_timers ();
00150 
00151         //  Wait for events.
00152         //  On Solaris, we can retrieve no more then (OPEN_MAX - 1) events.
00153         poll_req.dp_fds = &ev_buf [0];
00154 #if defined ZMQ_HAVE_SOLARIS
00155         poll_req.dp_nfds = std::min ((int) max_io_events, OPEN_MAX - 1);
00156 #else
00157         poll_req.dp_nfds = max_io_events;
00158 #endif
00159         poll_req.dp_timeout = timeout ? timeout : -1;
00160         int n = ioctl (devpoll_fd, DP_POLL, &poll_req);
00161         if (n == -1 && errno == EINTR)
00162             continue;
00163         errno_assert (n != -1);
00164 
00165         for (int i = 0; i < n; i ++) {
00166 
00167             fd_entry_t *fd_ptr = &fd_table [ev_buf [i].fd];
00168             if (!fd_ptr->valid || !fd_ptr->accepted)
00169                 continue;
00170             if (ev_buf [i].revents & (POLLERR | POLLHUP))
00171                 fd_ptr->reactor->in_event ();
00172             if (!fd_ptr->valid || !fd_ptr->accepted)
00173                 continue;
00174             if (ev_buf [i].revents & POLLOUT)
00175                 fd_ptr->reactor->out_event ();
00176             if (!fd_ptr->valid || !fd_ptr->accepted)
00177                 continue;
00178             if (ev_buf [i].revents & POLLIN)
00179                 fd_ptr->reactor->in_event ();
00180         }
00181     }
00182 }
