00049 void zmq::xpub_t::xread_activated (pipe_t *pipe_)
00050 {
00051     //  There are some subscriptions waiting. Let's process them.
00052     msg_t sub;
00053     sub.init ();
00054     while (true) {
00055 
00056         //  Grab next subscription.
00057         if (!pipe_->read (&sub)) {
00058             sub.close ();
00059             return;
00060         }
00061 
00062         //  Apply the subscription to the trie.
00063         unsigned char *data = (unsigned char*) sub.data ();
00064         size_t size = sub.size ();

00066         bool unique;
00067         if (*data == 0)
00068             unique = subscriptions.rm (data + 1, size - 1, pipe_);
00069         else
00070             unique = subscriptions.add (data + 1, size - 1, pipe_);
00071 
00072         //  If the subscription is not a duplicate store it so that it can be
00073         //  passed to used on next recv call.
00074         if (unique && options.type != ZMQ_PUB)
00075             pending.push_back (blob_t ((unsigned char*) sub.data (),
00076                 sub.size ()));
